from fractions import Fraction

# Create two fractions
a = Fraction(1, 3)
b = Fraction(1, 6)

# Perform operations
sum_result = a + b
print(f"{a} + {b} = {sum_result}")

difference = a - b
print(f"{a} - {b} = {difference}")

product = a * b
print(f"{a} * {b} = {product}")

quotient = a / b
print(f"{a} / {b} = {quotient}")
